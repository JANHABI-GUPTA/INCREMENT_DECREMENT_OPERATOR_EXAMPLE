# INCREMENT_DECREMENT_OPERATOR_EXAMPLE
USING JAVA LANGUAGE


class PrePost
{
   public static void main(String[] args)
   {
      int a=10;
      System.out.println(a);
      
      System.out.println("PrePost Incre...");
      System.out.println(a++);
      System.out.println(++a);
         
      System.out.println("PrePost Decre...");
      System.out.println(a--);
      System.out.println(--a);
   }
}
